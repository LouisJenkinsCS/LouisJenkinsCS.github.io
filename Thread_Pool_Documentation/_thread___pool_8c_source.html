140  // Set up the signal handler to pause.
141  struct sigaction pause_signal;
142  pause_signal.sa_handler = Pause_Handler;
143  pause_signal.sa_flags = SA_RESTART;
144  sigemptyset(&pause_signal.sa_mask);
145  if(sigaction(SIGUSR1, &pause_signal, NULL) == -1) fprintf(stderr, "Get_Task callback was unable to initialize a pause_handler\n");
146  self->is_setup = 1;
147  while(tp->keep_alive){
148  while(tp->keep_alive && !self->task) self->task = PBQueue_Timed_Dequeue(tp->queue, queue_timeout);
149  if(!tp->keep_alive) break;
150  TP_INCREMENT(tp->active_threads, tp->thread_count_change);
151  Process_Task(self);
152  TP_DECREMENT(tp->active_threads, tp->thread_count_change);
153  signal_if_queue_finished();

176 int Thread_Pool_Init(size_t number_of_threads){
177  // If the thread pool is already initialized, return.
178  if(tp) return 0;
179  // Instead of directly allocating the static thread pool, we make a temporary one and make the static thread pool point to it later.
180  // The reason is because once the thread pool is allocated, it opens up the opportunity for undefined behavior since it no longer is NULL.
181  Thread_Pool *temp_tp = malloc(sizeof(Thread_Pool));
182  temp_tp->keep_alive = 1;
183  temp_tp->paused = 0;
184  temp_tp->seconds_to_pause = 0;
185  temp_tp->thread_count = temp_tp->active_threads = 0;
186  temp_tp->queue = PBQueue_Create_Unbounded(compare_task_priority);
187  init_cond(&temp_tp->resume, NULL);
188  init_cond(&temp_tp->all_tasks_finished, NULL);
189  init_mutex(&temp_tp->thread_count_change, NULL);
190  init_mutex(&temp_tp->is_paused, NULL);
191  init_mutex(&temp_tp->no_tasks, NULL);
192  temp_tp->worker_threads = malloc(sizeof(pthread_t *) * number_of_threads);
193  tp = temp_tp;
194  pthread_attr_t attr;
195  pthread_attr_init(&attr);
196  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
197  int i = 0;
198  for(;i < number_of_threads; i++){
199  Worker *worker = malloc(sizeof(Worker));
200  worker->thread = malloc(sizeof(pthread_t));
201  worker->thread_id = i+1;
202  worker->is_setup = 0;
203  worker->task = NULL;
204  pthread_create(worker->thread, NULL, Get_Tasks, worker);
205  tp->worker_threads[i] = worker;
206  while(!tp->worker_threads[i]->is_setup) pthread_yield();
207  tp->thread_count++;
208  }
209  pthread_attr_destroy(&attr);
210  return 1;
211 }
